using System;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Привет, Юзер!");
            Console.WriteLine("Как тебя зовут?");
            string user_Name = Convert.ToString(Console.ReadLine());
            Console.WriteLine("Привет, {0}, сколько тебе лет?", user_Name);
            int user_Age = Convert.ToInt16(Console.ReadLine());
            if (user_Age > 24)
           
            {
                Console.WriteLine("Ого, {0}, да ты старпёр!", user_Name, user_Age);
            }
            else 
            {
                Console.WriteLine("Фи, {0}, ты ещё мелочь", user_Name);            
            }

            Console.ReadKey();

        }
    }
}
